# DFA token definitions for json files
#

# Whitespace including comments of form  # ....,   // ....,   /* .... */

WS: (   [\x00-\x20]+     |  \
        (//|\#) [\x00-\x7f^\n]*   |  \
        /\* ( \**[\x00-\x7f^/\*] |  [\x00-\x7f^\*] )* \*+ /  )+

COMMA: ,
COLON: :
LISTOPEN: \[
LISTCLOSE: \]
MAPOPEN: \{
MAPCLOSE: \}
ID:   \w[\w\d]*
FLOAT: [\+\-]?  (\d*\.\d+([eE][\+\-]?\d+)?|\d+)
STRING: "  (  [\x00-\x7f^\"\\]  |  \\[\x00-\x7f]  )*  "
BOOLEAN: (true|false)
NULL: null
